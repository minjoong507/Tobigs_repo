def column_null_ratio(df, top_num):
    df = (all_data.isnull().sum() / len(all_data)) * 100
    df = df.drop(df[df == 0].index).sort_values(ascending=False)[:top_num]
    missing_data = pd.DataFrame({'Missing Ratio (%)' : df})
    display(missing_data.head(top_num))
    
def check_data(data):        
    with pd.option_context('display.max_rows', None, 'display.max_columns', None):
        print('--- column headers ----')
        print(data.columns.values) 
        print('-----------------------')

        print('-- column data types --')
        print(data.dtypes)
        print('-----------------------')

        print('-- column statistics --')
        print(data.describe()) 
        print('-----------------------')

        print('-- column null count --')
        print(data.isnull().sum())
        print('-----------------------')
        
        print('-- column null ratio --')
        print(column_null_ratio(data, 10))
        print('-----------------------')
